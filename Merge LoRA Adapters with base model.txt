import torch
from transformers import AutoTokenizer, AutoModelForCausalLM
from peft import PeftModel
import os

# Configuration
BASE_MODEL = "meta-llama/Llama-3.2-1B-Instruct"
ADAPTER_PATH = "Output"  # Your fine-tuned LoRA adapters
MERGED_OUTPUT = "Merged"

print("🔗 Merging LoRA adapters with base model...")
print(f"📂 Base model: {BASE_MODEL}")
print(f"📂 Adapters: {ADAPTER_PATH}")
print(f"📁 Output: {MERGED_OUTPUT}")
print("="*70)

# Create output directory
os.makedirs(MERGED_OUTPUT, exist_ok=True)

# Load tokenizer
print("\n📥 Loading tokenizer...")
tokenizer = AutoTokenizer.from_pretrained(ADAPTER_PATH)

# Load base model
print("📥 Loading base model...")
base_model = AutoModelForCausalLM.from_pretrained(
    BASE_MODEL,
    torch_dtype=torch.float16,
    device_map="auto",
    low_cpu_mem_usage=True
)

# Load LoRA adapters
print("🔧 Loading LoRA adapters...")
model = PeftModel.from_pretrained(base_model, ADAPTER_PATH)

# Merge LoRA weights into base model
print("⚡ Merging LoRA weights into base model...")
model = model.merge_and_unload()

# Save merged model
print(f"💾 Saving merged model to {MERGED_OUTPUT}...")
model.save_pretrained(
    MERGED_OUTPUT,
    safe_serialization=True,  # Use safetensors format
    max_shard_size="2GB"
)
tokenizer.save_pretrained(MERGED_OUTPUT)

print("✅ Model merged successfully!")
print(f"📁 Merged model saved to: {MERGED_OUTPUT}")
print("="*70)

# Quick test
print("\n🧪 Testing merged model...")
test_question = """Please answer the following multiple-choice question:
A patient presents with high fever and cough. What is the initial assessment?
A. ECG
B. Chest X-ray
C. Blood test
D. CT scan"""

messages = [{"role": "user", "content": test_question}]

inputs = tokenizer.apply_chat_template(
    messages,
    add_generation_prompt=True,
    tokenize=True,
    return_tensors="pt",
).to(model.device)

model.eval()
with torch.no_grad():
    outputs = model.generate(
        inputs,
        max_new_tokens=256,
        do_sample=True,
        temperature=0.7,
        top_p=0.9,
        pad_token_id=tokenizer.eos_token_id
    )

response = tokenizer.decode(outputs[0][inputs.shape[-1]:], skip_special_tokens=True)

print("\n" + "="*70)
print("TEST RESPONSE:")
print("="*70)
print(response)
print("="*70)

print("\n🎉 Merged model is ready for GGUF conversion!")
